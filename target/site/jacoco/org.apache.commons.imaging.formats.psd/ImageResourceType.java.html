    public static String getDescription(final int id) {
        for (final ImageResourceType type : TYPES) {
            if (type.from <= id && id <= type.to) {
                return type.description;
            }
        }
        return "Unknown";
    }

    public final int from;
    public final int to;

    public final String description;

    ImageResourceType(final int id, final int id2, final String description) {
        this.from = id;
        this.to = id2;
        this.description = description;
    }

    ImageResourceType(final int id, final String description) {
        this.from = id;
        this.to = id;
        this.description = description;
    }
